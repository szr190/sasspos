Load By RequireJS组件基础使用

1、页面引入js、css：

<link rel="stylesheet" href="demo.css">

<script src="http://requirejs.org/docs/release/2.1.17/minified/require.js">

2、html标签：

<input type="text" name="user[email]" data-rule="required;email" placeholder="Email">

<input type="password" name="password" data-rule="required;" placeholder="Password"><input type="text" name="user[email]" data-rule="required;email" placeholder="Email">

<button type="submit">Submit</button>

3、组件渲染：

<script>

requirejs.config({

paths: {

jquery: 'http://dn-nice.qbox.me/libs/jquery/1.x/jquery.min',

validator: '../local/en'

}

});

define('demo', ['jquery', 'validator'], function($){

});

require(['demo']);
